        show background rectangle]
        \Vertex[x=0,y=1]{1};
        \Vertex[x=0,y=0]{2};
        \Vertex[x=1,y=0]{3};
        \Loop[dist=0.8cm](1)
        \Edge[style={bend left,->}](1)(3);
        \tikzset{EdgeStyle/.style={bend right,->}}
        \Edge(1)(2);
        \Edge(2)(1);
        \Edge(2)(3);
      \end{tikzpicture}
    \end{column}
  \end{columns}
  \[
    (V_1, E_1) \connect (V_2, E_2) \coloneqq (V_1 \cup V_2, E_1 \cup E_2 \cup
    V_1 \times V_2)
  \] 
\end{frame}
